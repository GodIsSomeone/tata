/*
输入一个链表，反转链表后，输出新链表的表头。
*/

```
/*
struct ListNode {
	int val;
	struct ListNode *next;
	ListNode(int x) :
			val(x), next(NULL) {
	}
};*/
class Solution {
public:
    ListNode* ReverseList(ListNode* head) {
        if (head == NULL || head->next ==NULL)
        {
            return head;
        }
        ListNode *preNode = NULL;
        ListNode *curNode = head;
        while (curNode)
        {
            ListNode* nextNode = curNode->next;
            
            curNode->next = preNode;
            preNode = curNode;
            curNode = nextNode;
        }
        return preNode;
    }
};
```
