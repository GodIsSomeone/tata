/*
输入两个单调递增的链表，输出两个链表合成后的链表，当然我们需要合成后的链表满足单调不减规则。
*/

```
/*
struct ListNode {
	int val;
	struct ListNode *next;
	ListNode(int x) :
			val(x), next(NULL) {
	}
};*/
class Solution {
public:
    ListNode* Merge(ListNode* pHead1, ListNode* pHead2)
    {
        if (pHead1 == NULL)
        {
            return pHead2;
        }
        if (pHead2 == NULL)
        {
            return pHead1;
        }
        ListNode* retList;
        ListNode* tmpNode;
	
	/*确定头结点*/
        if (pHead1->val < pHead2->val)
        {
            retList = pHead1;
            pHead1 = pHead1->next;
        }
        else
        {
            retList = pHead2;
            pHead2 = pHead2->next;
        }
        tmpNode = retList;
        while(pHead1&&pHead2)
        {
            if (pHead1->val < pHead2->val)
            {
                tmpNode->next = pHead1;
                pHead1 = pHead1->next;
            }
            else
            {
                tmpNode->next = pHead2;
                pHead2 = pHead2->next;
            }
            tmpNode = tmpNode->next;
        }
        /*其中一个结点走完了，直接赋值即可*/
        if (!pHead1)
        {
            tmpNode->next = pHead2;
        }
        else
        {
            tmpNode->next = pHead1;
        }
        return retList;
    }
};
```
